425   def universal_saldo(algorithm = :default, with_subcategories = false, additional_condition="", *params)
426     algorithm = algorithm(algorithm, with_subcategories)
427     unless additional_condition.blank?
428       algorithm[:conditions].first << " AND #{additional_condition}"
429       algorithm[:conditions] += params
430     end
431 
432     money = Money.new()
433 
434     TransferItem.sum(:value, algorithm).each do |set|
435       if set.class == Array
436         # group by currency
437         currency, value = set
438         currency = Currency.find_by_id(currency)
439         money.add!(value.round(2), currency)
440       else
441         # calculated to one value in default currency
442         money.add!(set.to_f.round(2), Currency.find_by_id(self.user.default_currency))
443       end
444     end
445 
446     if self.user.invert_saldo_for_income && self.category_type == :INCOME
447       money = Money.new - money
448     end
449 
450     return money
