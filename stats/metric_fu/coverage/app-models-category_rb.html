388   def universal_saldo(algorithm = :default, with_subcategories = false, additional_condition="", *params)
389     algorithm = algorithm(algorithm, with_subcategories)
390     unless additional_condition.blank?
391       algorithm[:conditions].first << " AND #{additional_condition}"
392       algorithm[:conditions] += params
393     end
394 
395     money = Money.new()
396 
397     TransferItem.sum(:value, algorithm).each do |set|
398       if set.class == Array
399         # group by currency
400         currency, value = set
401         currency = Currency.find_by_id(currency)
402         money.add(value, currency)
403       else
404         # calculated to one value in default currency
405         money.add(set.to_f, Currency.find_by_id(self.user.default_currency))
406       end
407     end
408     return money
