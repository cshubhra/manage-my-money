361   def get_y_axis_for_report(min,max, right = false)
362     
363     y_axis = if right
364         YAxisRight.new
365       else
366         YAxis.new
367     end
368 
369     distance = (max - min).abs
370     distance = 10 if distance < 10
371     step = 10**( ( Math.log10( distance ) ).ceil() -1)
372 
373     steps = distance.to_f/step.to_f
374 
375     step /= case steps
376     when 1...3 then 4
377     when 3..6 then 2
378     else 1
379     end
380 
381     liczba = max.abs / step
382     liczba = liczba.ceil if max >= 0
383     liczba = liczba.floor if max < 0
384     max_max = liczba * step
385     max_max *= -1 if max < 0
386 
387     liczba = min.abs / step
388     liczba = liczba.floor if min >= 0
389     liczba = liczba.ceil if min < 0
390     min_min = liczba * step
391     min_min *= -1 if min < 0
392 
393     y_axis.set_range(min_min, max_max, step)
394     y_axis
395   end
